常常需要捕捉 用户的输入后用回车提交的事件，而不是用鼠标单击
一般有两种情况

一、整个页面用一个回车提交事件：

        document.onkeydown = function(evt){
			var evt=evt?evt:(window.event?window.event:null);
				if((evt.keyCode || evt.which) == 13){
				alert('整个页面中你按了回车')
				}
        }

二、某个输入框中注册回车事件，一般用于用户在页面输入后按回车:



        function enterIn(evt){
			var evt=evt?evt:(window.event?window.event:null);//兼容IE和FF
			if (evt.keyCode==13){
			   alert('input1你按回车了')
			}
        }
		document.getElementById('input1').onkeydown=function(event){enterIn(event)}
		
		
        function enterIn2(evt){
			var evt=evt?evt:(window.event?window.event:null);//兼容IE和FF
			if (evt.keyCode==13){
			    alert('input2你按回车了')
			}
        }
		document.getElementById('input2').onkeydown=function(event){enterIn2(event)}

如果一个页面可能需要捕捉多个回车事件，那么就利用上面的方式分开写>
