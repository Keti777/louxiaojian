          function lxj(){
              var mods = [].slice.call(arguments, 0); //将函数的形参储存为数组
              return mods;
          }
          document.write(lxj(1,2)+'
');//输出结果为:[1,2]
          document.write(lxj(1,2,3)+'
');//输出结果为:[1,2,3]
          document.write(lxj('a','b')+'
');//输出结果为:['a','b']
          document.write(lxj('a','b','h','j')+'
');//输出结果为:['a','b','h','j']
          document.write(lxj('a','b','y')+'
');//输出结果为:['a','b','y']
