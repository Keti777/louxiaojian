          function stylename(name){
              var RE_DASH = /-([a-z])/ig;
              var CAMELCASE_FN = function(all, letter) {
                  return letter.toUpperCase();
              };
              if (name.indexOf('-') > 0) {
                  // webkit 认识 camel-case, 其它内核只认识 cameCase
                  name = name.replace(RE_DASH, CAMELCASE_FN);
              }
              return name;
          }
