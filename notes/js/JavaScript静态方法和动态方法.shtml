function JObj(){
	this.author = "xling";
	this.aa = function(pParm){
	   alert(pParm);
    }
    var bb = function(pParm){
       alert(pParm);
    }
}

JObj.name='lou';
JObj.cc = function(pParm){
       alert(pParm);
}

JObj.prototype.dd = function(pParm){
      alert(pParm);
}

document.write("
===========以下是Class中的，即所谓的静态，不能由实例调用，只能用类名调用================
");

for(var o in JObj){
document.write(o," = ",eval("JObj." + o),"
");
}
JObj.cc("123")
document.write("
===========以下是实例中的，只能用实例来调用================
");

var obj = new JObj();
for(var o in obj){
	document.write(o," = ",eval("obj." + o),"
");
}
obj.cc("123")
