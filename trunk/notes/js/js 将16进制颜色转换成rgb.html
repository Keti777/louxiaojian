function hex2rgb(a) //将16进制颜色转换成rgb
{
	a=a.substring(1);
	a=a.toLowerCase();
	b=new Array();
	for(x=0;x<3;x++){
  		b[0]=a.substr(x*2,2);
  		b[3]="0123456789abcdef";
  		b[1]=b[0].substr(0,1);
   		b[2]=b[0].substr(1,1);
		b[20+x]=b[3].indexOf(b[1])*16+b[3].indexOf(b[2]);
	}
	return "rgb(" +b[20]+","+b[21]+","+b[22]+ ")";
}
