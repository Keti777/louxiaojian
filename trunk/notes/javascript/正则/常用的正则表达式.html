  2. 将一个CSS属性转换为驼峰形式，例如(font-size --> fontSize)

                  function stylename(name){
                  var RE_DASH = /-([a-z])/ig;
                  var CAMELCASE_FN = function(all, letter) {
                      return letter.toUpperCase();
                  };
                  if (name.indexOf('-') > 0) {
                      // webkit 认识 camel-case, 其它内核只认识 cameCase
                      name = name.replace(RE_DASH, CAMELCASE_FN);
                  }
                  return name;
              }
              document.write('background-color——>'+stylename('background-color')+'
    ');
              document.write('border-bottom-width——>'+stylename('border-bottom-width'))

                strtrim=_trim ?  ((str == undef) ? EMPTY : _trim.call(str)) : ((str == undef) ? EMPTY : str.toString().replace(RE_TRIM, EMPTY));
