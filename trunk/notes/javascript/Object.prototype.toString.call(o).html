Object.prototype.toString.call(o)

           function optc(o){
               return Object.prototype.toString.call(o)
           }
           document.write(optc([1,2,3])+'
');//输出[object Array]
           document.write(optc(function(){})+'
');//输出[object Function]
           document.write(optc({})+'
');//输出[object Object]
           document.write(optc(1)+'
');//输出[object Number]
           document.write(optc('asdf')+'
');//输出[object String]
           document.write(optc(new Date())+'
');//输出[object Date]
           document.write(optc(1==1)+'
');//输出[object Boolean]
